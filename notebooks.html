# Proximity Measure for Nominal Attributes
def d1(i,j):
    P = 2
    Q = 0
    for col in ["Address","Zip"]:
        if dt[col][i] == dt[col][j]:
            Q += 1
    return (P-Q)/P

print(d1(1,2))
print(d1(1,3))
print(d1(1,4))

# Distance on Numeric Data: Minkowski Distance
def d2(i,j, h):
    result = 0
    for col in ["Price","Area","Room","Lon","Lat"]:
        result = math.pow(abs(dt[col][i] - dt[col][j]), h)
    return result**(1/h)

# Manhattan distance
print(d2(1,2,1))
print(d2(1,3,1))
print(d2(1,4,1))
